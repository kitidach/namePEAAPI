1. ค้นหาสำนักงาน

GET /offices ค้นหาสำนักงานทั้งหมด พร้อม filter & pagination

ค้นหาสำนักงาน กรองตามขนาด กฟจ. หรือจุดรวมงาน รองรับ pagination และ sorting

Parameters

ชื่อค่าคำอธิบาย
q optionalstringคำค้น: รหัส, ชื่อย่อ, ชื่อเต็ม
sizeL, M, S, XSกรองตามขนาด
group8stringกรองตาม กฟจ. เช่น G8_01
groupCodestringกรองตามจุดรวมงาน เช่น D01
pagenumberหน้า (default: 1)
limitnumberจำนวนต่อหน้า (default: 50, max: 200)
sortcode, shortName, fullName, size, indexเรียงตาม
orderasc, descลำดับ (default: asc)

ตัวอย่าง

curl "https://namepeaapi.pages.dev/api/v1/offices?q=อุดร&size=L"
const res = await fetch(
  "https://namepeaapi.pages.dev/api/v1/offices?q=อุดร&limit=10"
);
const json = await res.json();
console.log(json.data); // [{code, shortName, fullName, ...}]
import requests

r = requests.get("https://namepeaapi.pages.dev/api/v1/offices",
    params={"q": "อุดร", "size": "L"})
data = r.json()
for office in data["data"]:
    print(office["shortName"], office["fullName"])
function getOffices() {
  var url = "https://namepeaapi.pages.dev/api/v1/offices?q=อุดร";
  var res = UrlFetchApp.fetch(url);
  var json = JSON.parse(res.getContentText());

  var sheet = SpreadsheetApp.getActiveSheet();
  json.data.forEach(function(o, i) {
    sheet.getRange(i + 1, 1).setValue(o.code);
    sheet.getRange(i + 1, 2).setValue(o.shortName);
    sheet.getRange(i + 1, 3).setValue(o.fullName);
    sheet.getRange(i + 1, 4).setValue(o.size);
  });
}

2. รายละเอียดสำนักงาน

GET /offices/{code} ดึงข้อมูลสำนักงาน 1 แห่งจากรหัส

ดึงข้อมูลสำนักงานจาก code เช่น D01101

curl "https://namepeaapi.pages.dev/api/v1/offices/D01101"
function getOfficeByCode(code) {
  var url = "https://namepeaapi.pages.dev/api/v1/offices/" + code;
  var res = UrlFetchApp.fetch(url);
  var json = JSON.parse(res.getContentText());
  Logger.log(json.data.shortName + " — " + json.data.fullName);
  return json.data;
}

3. จัดกลุ่มสำนักงาน ⭐

GET /tools/group-offices จัดกลุ่มสำนักงานเป็น กฟจ.(...), กฟส.(...)

กรอกชื่อย่อ/ชื่อเต็ม/รหัส → จัดกลุ่มเป็นรูปแบบ กฟจ.(...), กฟส.(...) พร้อม copy ได้เลย

💡 รองรับหลาย format: กฟส.รณ., เรณูนคร, รณ., D06202

Parameters

ชื่อค่าคำอธิบาย
names requiredstringชื่อสำนักงาน คั่นด้วย , เช่น กฟส.รณ.,กฟส.ภผม.

ตัวอย่าง

# ชื่อย่อ
curl "https://namepeaapi.pages.dev/api/v1/tools/group-offices?names=กฟส.รณ.,กฟส.ภผม."

# ชื่อเต็ม
curl "https://namepeaapi.pages.dev/api/v1/tools/group-offices?names=เรณูนคร,ภูผาม่าน"

# Response:
# {
#   "data": {
#     "formatted": "กฟจ.(นพ.,ขก.),\nกฟส.(ธพ.,ชมพ.,รณ.,ภผม.)"
#   }
# }
async function groupOffices(names) {
  const url = new URL("https://namepeaapi.pages.dev/api/v1/tools/group-offices");
  url.searchParams.set("names", names);

  const res = await fetch(url);
  const json = await res.json();
  return json.data.formatted;
  // "กฟจ.(นพ.,ขก.),\nกฟส.(ธพ.,ชมพ.,รณ.,ภผม.)"
}

// ใช้งาน
const result = await groupOffices("กฟส.รณ.,กฟส.ภผม.,กฟส.คตก.");
console.log(result);
/**
 * จัดกลุ่มสำนักงาน — ใช้ใน Google Sheets ได้เลย
 * =groupPEA("กฟส.รณ.,กฟส.ภผม.")
 */
function groupPEA(names) {
  var url = "https://namepeaapi.pages.dev/api/v1/tools/group-offices?names="
            + encodeURIComponent(names);
  var res = UrlFetchApp.fetch(url);
  var json = JSON.parse(res.getContentText());
  return json.data.formatted;
}

// ตัวอย่างผลลัพธ์:
// กฟจ.(นพ.,ขก.),
// กฟส.(ธพ.,ชมพ.,รณ.,ภผม.)

6. Export CSV / JSON

GET /export/offices ดาวน์โหลดข้อมูลทั้งหมดเป็น CSV หรือ JSON
ชื่อค่าคำอธิบาย
format requiredcsv, jsonรูปแบบไฟล์
q, size, group8, groupCodestringfilter เหมือน /offices
 
# CSV (Excel-compatible, UTF-8 BOM)
curl -o offices.csv "https://namepeaapi.pages.dev/api/v1/export/offices?format=csv"

# JSON
curl -o offices.json "https://namepeaapi.pages.dev/api/v1/export/offices?format=json"

# พร้อม filter
curl -o filtered.csv "https://namepeaapi.pages.dev/api/v1/export/offices?format=csv&size=L"
import requests
import csv
import io

r = requests.get("https://namepeaapi.pages.dev/api/v1/export/offices",
    params={"format": "csv"})
r.encoding = "utf-8-sig"

reader = csv.DictReader(io.StringIO(r.text))
for row in reader:
    print(row["รหัส"], row["ชื่อย่อ"])

Response Format

ทุก endpoint ตอบ JSON ในรูปแบบเดียวกัน:

{
  "success": true,
  "data": { ... },        // ข้อมูล
  "meta": {               // pagination / stats
    "total": 116,
    "page": 1,
    "limit": 50
  },
  "error": null            // null ถ้าสำเร็จ
}

// กรณี error:
{
  "success": false,
  "data": null,
  "error": {
    "code": "NOT_FOUND",
    "message": "Office not found: XXX"
  }
}

⚡ ข้อจำกัดและหมายเหตุ

  • ✅ ใช้ฟรี ไม่ต้อง API key หรือ login
  • ✅ รองรับ CORS — เรียกจาก browser ได้เลย
  • ✅ CSV มี UTF-8 BOM — เปิดใน Excel ภาษาไทยได้
  • 📦 ข้อมูลเป็น กฟฉ.1 (116 สำนักงาน)
  • ⚠️ ไม่ควรเรียก API เกิน 100 requests/นาที
  • 🔄 ข้อมูลอัปเดตเมื่อ deploy ใหม่ ไม่ใช่ real-time
